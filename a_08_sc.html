def S1(t):
    if t in diff_cache1:
        return diff_cache1[t]
    if t == 0:
        value = 0
    elif t >0:
        value = 1+S1(t-1)*(1+r0)-c*((1+z1)**t)
    diff_cache1[t] = value
    return value
